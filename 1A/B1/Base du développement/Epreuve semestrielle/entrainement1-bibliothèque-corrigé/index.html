    Corrigé : Application de gestion d'une bibliothèque

    Exercice 1 : Déclaration de variables et types de données

    Ici, l'objectif est de choisir le bon type (entier ou chaîne de caractères) et d'initialiser correctement.

    // 1. Déclaration des variables
int nbLivresTotal;
string titreLivre;
string nomEmprunteur;
int dureeEmprunt;

// 2. Assignation des valeurs initiales
nbLivresTotal = 100;
titreLivre = "";
// Le nom et la durée seront définis plus tard par l'utilisateur

    Exercice 2 : Conditions et prise de décision

    L'objectif est d'interagir avec l'utilisateur et d'utiliser une structure conditionnelle simple (if/else).

    Console.Write("Quel livre souhaitez-vous emprunter ? ");
string saisieLivre = Console.ReadLine();

// Vérification de la disponibilité
if (nbLivresTotal > 0)
{
    Console.WriteLine($"Le livre \"{saisieLivre}\" est disponible. Votre emprunt est confirmé.");
}
else
{
    Console.WriteLine("Désolé, aucun exemplaire n'est disponible pour le moment.");
}

    Exercice 3 : Boucles et validation des entrées

    On sécurise la saisie pour éviter les chaînes vides. La boucle do...while est idéale car on doit demander au moins une fois.

    string titreValide;

do
{
    Console.Write("Quel livre souhaitez-vous emprunter ? ");
    titreValide = Console.ReadLine();

    if (string.IsNullOrWhiteSpace(titreValide))
    {
        Console.WriteLine("Erreur : Le titre ne peut pas être vide. Veuillez réessayer.");
    }

} while (string.IsNullOrWhiteSpace(titreValide));

Console.WriteLine($"Titre validé : {titreValide}");

    Exercice 4 : Utilisation des tableaux et des listes

    On distingue ici le Tableau (taille fixe, données statiques) de la Liste (taille dynamique, données évolutives).

    // 1. Tableau de livres populaires
string[] livresPopulaires = { "1984", "Le Petit Prince", "L'Étranger", "Dune", "Fondation" };

Console.WriteLine("--- Livres populaires ---");
// 2. Affichage avec foreach
foreach (string livre in livresPopulaires)
{
    Console.WriteLine("- " + livre);
}

// 3. Liste dynamique pour les emprunts
List<string> livresEmpruntes = new List<string>();

// Simulation d'un ajout (récupéré depuis l'exercice précédent)
Console.WriteLine("\nAjout du livre sélectionné à la liste des emprunts...");
livresEmpruntes.Add(titreValide); // On utilise la variable de l'Ex 3

// 4. Affichage de la liste mise à jour
Console.WriteLine("--- Liste des livres empruntés ---");
foreach (string emprunt in livresEmpruntes)
{
    Console.WriteLine($"> {emprunt}");
}

    Exercice 5 : Création de fonctions

    On encapsule la logique de calcul de date. L'utilisation de DateTime est un classique en gestion.

    // Déclaration de la fonction (à placer à l'intérieur de la classe Program, mais hors du Main)
static DateTime CalculerDateRetour(int duree)
{
    DateTime dateDuJour = DateTime.Now;
    return dateDuJour.AddDays(duree);
}

// Appel de la fonction (dans le Main)
// Test avec 10 jours
DateTime dateRetourTest = CalculerDateRetour(10);
Console.WriteLine($"Si vous empruntez aujourd'hui pour 10 jours, retour le : {dateRetourTest:dd/MM/yyyy}");

    Exercice 6 : Mise en application complète (Code final)

    // Variable globale pour le stock (pour simplifier l'exercice)
static int nbLivresTotal = 100;

// Liste pour stocker l'historique des emprunts de la session
static List<string> listeEmprunts = new List<string>();

static void Main(string[] args)
{
    Console.WriteLine("=== APPLICATION DE GESTION BIBLIOTHÈQUE ===\n");

    // --- 1. Saisie des données ---

    // A. Saisie du Nom
    string nomEmprunteur;
    do
    {
        Console.Write("Nom de l'emprunteur : ");
        nomEmprunteur = Console.ReadLine();
    } while (string.IsNullOrWhiteSpace(nomEmprunteur));

    // B. Saisie du Titre
    string titreLivre;
    do
    {
        Console.Write("Titre du livre à emprunter : ");
        titreLivre = Console.ReadLine();
    } while (string.IsNullOrWhiteSpace(titreLivre));

    // C. Saisie de la Durée (avec validation que c'est bien un nombre)
    int dureeEmprunt = 0;
    bool dureeValide = false;
    while (!dureeValide)
    {
        Console.Write("Durée de l'emprunt (en jours) : ");
        string saisieDuree = Console.ReadLine();

        // TryParse évite que le programme plante si l'utilisateur tape "douze" au lieu de "12"
        if (int.TryParse(saisieDuree, out dureeEmprunt) && dureeEmprunt > 0)
        {
            dureeValide = true;
        }
        else
        {
            Console.WriteLine("Veuillez entrer un nombre entier positif valide.");
        }
    }

    // --- 2. Traitement ---

    if (nbLivresTotal > 0)
    {
        // Réduire le stock
        nbLivresTotal--; 

        // Ajouter à la liste
        listeEmprunts.Add(titreLivre);

        // Calcul de la date
        DateTime dateRetour = CalculerDateRetour(dureeEmprunt);

        // --- 3. Affichage du résumé ---
        Console.WriteLine("\n---------------------------------");
        Console.WriteLine("       RÉSUMÉ DE L'EMPRUNT       ");
        Console.WriteLine("---------------------------------");
        Console.WriteLine($"Nom de l'emprunteur   : {nomEmprunteur}");
        Console.WriteLine($"Livre emprunté        : {titreLivre}");
        Console.WriteLine($"Durée de l'emprunt    : {dureeEmprunt} jours");
        // Le format :d affiche la date courte (ex: 10/12/2024)
        Console.WriteLine($"Date de retour prévue : {dateRetour:d}"); 
        Console.WriteLine($"Livres restants       : {nbLivresTotal}");
        Console.WriteLine("---------------------------------");
    }
    else
    {
        Console.WriteLine("\nErreur : Stock épuisé, impossible d'emprunter.");
    }

    // Pause pour lire le résultat
    Console.WriteLine("\nAppuyez sur une touche pour quitter...");
    Console.ReadKey();
}

// Fonction de l'exercice 5
static DateTime CalculerDateRetour(int duree)
{
    return DateTime.Now.AddDays(duree);
}
